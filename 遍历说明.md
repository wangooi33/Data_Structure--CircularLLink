void CLList_Traverse(CLNode_t* Head) {
    if (!Head || Head->next == NULL) return;

    CLNode_t* current = Head->next;
    CLNode_t* start = current;

    while (current != NULL) {
        printf("%d -> ", current->data);
        current = current->next;

        if (current == start) break;
    }
    printf("(head)\n");
}
